## 3. DATABASE SCHEMA UPDATES

### 3.1 New Tables to Create

#### `budgets` Table
```typescript
{
  id: string (primary key)
  project_id: string (foreign key → projects)
  // NO site_id - PROJECT LEVEL ONLY per user requirement
  category: string ('material' | 'labor' | 'equipment' | 'other')
  allocated_amount: number
  description: string
  created_by: string (user_id)
  created_at: number (timestamp)
  updated_at: number (timestamp)
  sync_status: string
  _version: number
}
```

#### `costs` Table
```typescript
{
  id: string (primary key)
  project_id: string (foreign key → projects)
  // NO site_id - PROJECT LEVEL ONLY per user requirement
  po_id: string (foreign key → purchase_orders, nullable)
  category: string ('material' | 'labor' | 'equipment' | 'other')
  amount: number
  description: string
  cost_date: number (timestamp)
  created_by: string (user_id)
  created_at: number (timestamp)
  updated_at: number (timestamp)
  sync_status: string
  _version: number
}
```

#### `invoices` Table
```typescript
{
  id: string (primary key)
  project_id: string (foreign key → projects)
  po_id: string (foreign key → purchase_orders)
  invoice_number: string
  invoice_date: number (timestamp)
  amount: number
  payment_status: string ('pending' | 'paid')
  payment_date: number (timestamp, nullable)
  vendor_id: string (foreign key → vendors)
  created_by: string (user_id)
  created_at: number (timestamp)
  updated_at: number (timestamp)
  sync_status: string
  _version: number
}
```

#### `po_items` Table (Line items for POs)
```typescript
{
  id: string (primary key)
  po_id: string (foreign key → purchase_orders)
  item_id: string (foreign key → items, nullable)
  description: string
  quantity: number
  unit_price: number
  total_price: number
  created_at: number (timestamp)
  updated_at: number (timestamp)
}
```
